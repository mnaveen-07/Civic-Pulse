<script>
(function() {
    'use strict';
    
    const API_BASE_URL = 'http://localhost:5000/api';
    const getAuthToken = () => localStorage.getItem('authority_token');
    const setAuthToken = (token) => localStorage.setItem('authority_token', token);
    const clearAuthToken = () => localStorage.removeItem('authority_token');

    const apiCall = async (endpoint, options = {}) => {
        const token = getAuthToken();
        const headers = {
            'Content-Type': 'application/json',
            ...(token && { 'Authorization': `Bearer ${token}` }),
            ...options.headers
        };

        try {
            const response = await fetch(`${API_BASE_URL}${endpoint}`, {
                ...options,
                headers
            });

            if (!response.ok) {
                const error = await response.json();
                throw new Error(error.error || 'API request failed');
            }

            return await response.json();
        } catch (error) {
            console.error('API Error:', error);
            throw error;
        }
    };

    const API = {
        authority: {
            login: (credentials) => apiCall('/authority/login', { method: 'POST', body: JSON.stringify(credentials) }),
            getIssues: (status = 'all') => apiCall(`/authority/issues?status=${status}`),
            updateStatus: (id, data) => apiCall(`/authority/issues/${id}/status`, { method: 'PUT', body: JSON.stringify(data) }),
            getStats: () => apiCall('/authority/stats'),
            getAnalytics: () => apiCall('/authority/analytics')
        }
    };

    const appState = {
        currentAuthority: null,
        issues: [],
        stats: {},
        map: null,
        charts: {}
    };

    const DOMElements = {
        loginContainer: document.getElementById('login-container'),
        appWrapper: document.getElementById('app-wrapper'),
        loginForm: document.getElementById('login-form'),
        logoutBtn: document.getElementById('logoutBtn'),
        issuesTableBody: document.getElementById('issues-table-body'),
        resolvedTableBody: document.getElementById('resolved-table-body'),
        navLinks: document.querySelectorAll('.sidebar nav a'),
        views: document.querySelectorAll('.main-content .view'),
        modalContainer: document.getElementById('modal-container'),
        criticalCount: document.getElementById('critical-count'),
        urgentCount: document.getElementById('urgent-count'),
        totalCount: document.getElementById('total-count'),
        resolvedCount: document.getElementById('resolved-count'),
        darkModeToggle: document.getElementById('dark-mode-toggle'),
        officialName: document.getElementById('official-name'),
        toastContainer: document.getElementById('toast-container'),
        severityFilter: document.getElementById('severity-filter')
    };

    // Get severity label
    const getSeverityLabel = (severity) => {
        const labels = ['Minor', 'Moderate', 'Severe', 'Critical'];
        return labels[severity] || 'Unknown';
    };

    function showToast(message, type = 'success') {
        const toast = document.createElement('div');
        toast.className = `toast ${type}`;
        toast.textContent = message;
        DOMElements.toastContainer.appendChild(toast);
        setTimeout(() => toast.classList.add('show'), 10);
        setTimeout(() => {
            toast.classList.remove('show');
            setTimeout(() => toast.remove(), 300);
        }, 3000);
    }

    function calculatePriority(issue) {
        const categoryWeights = { 
            "Public Safety": 1.5, "Sanitation": 1.2, "Water Supply": 1.1, 
            "Roads": 1.0, "Electricity": 0.9, "Other": 0.7 
        };
        const score = (issue.upvotes * 1) + (categoryWeights[issue.category] || 1.0) * 50;
        return Math.round(score);
    }

    function getPriorityLevel(score) {
        if (score > 200) return { label: 'Critical', class: 'priority-critical' };
        if (score > 100) return { label: 'Urgent', class: 'priority-urgent' };
        return { label: 'Routine', class: 'priority-routine' };
    }

    function populateTable(tbody, issues) {
        tbody.innerHTML = '';
        if (issues.length === 0) {
            tbody.innerHTML = '<tr><td colspan="7" style="text-align:center; color: var(--text-light);">No issues found</td></tr>';
            return;
        }
        issues.forEach(issue => {
            const priorityScore = calculatePriority(issue);
            const priority = getPriorityLevel(priorityScore);
            const severity = issue.severity || 1;
            const isCritical = issue.is_critical || severity === 3;
            
            const row = document.createElement('tr');
            if (isCritical) {
                row.style.backgroundColor = 'rgba(124, 58, 237, 0.1)';
            }
            
            row.innerHTML = `
                <td><span class="priority-dot ${priority.class}"></span>${priority.label} (${priorityScore})</td>
                <td>${issue.title}</td>
                <td>${issue.category}</td>
                <td>
                    <div class="severity-indicator">
                        <span class="severity-pill severity-${severity}">${getSeverityLabel(severity)}</span>
                        ${isCritical ? '<i class="fas fa-exclamation-triangle" style="color: var(--danger-color);"></i>' : ''}
                    </div>
                </td>
                <td>${new Date(issue.created_at || issue.date).toLocaleDateString()}</td>
                <td>${issue.status}</td>
                <td><button class="btn view-details" data-issue-id="${issue._id || issue.id}">Details</button></td>`;
            tbody.appendChild(row);
        });
    }

    function openFilteredIssueListModal(filter) {
        let issuesToShow, title;
        const issuesWithPriority = appState.issues.map(issue => ({ 
            ...issue, 
            priorityScore: calculatePriority(issue) 
        }));
        const activeIssues = issuesWithPriority.filter(i => i.status !== 'Resolved');

        switch (filter) {
            case 'critical':
                issuesToShow = activeIssues.filter(i => i.severity === 3 || i.is_critical);
                title = "Critical Issues";
                break;
            case 'urgent':
                issuesToShow = activeIssues.filter(i => i.severity === 2);
                title = "Severe Issues";
                break;
            case 'active':
                issuesToShow = activeIssues;
                title = "All Active Issues";
                break;
            default:
                issuesToShow = [];
                title = "Issues";
        }
        
        const modalHTML = `
            <div class="modal-overlay">
                <div class="modal-content">
                    <div class="modal-header">
                        <h2>${title}</h2>
                        <button class="modal-close">&times;</button>
                    </div>
                    <div class="modal-body">
                        <div class="table-wrapper">
                            <table>
                                <thead>
                                    <tr>
                                        <th>Priority</th>
                                        <th>Issue</th>
                                        <th>Category</th>
                                        <th>Severity</th>
                                        <th>Upvotes</th>
                                        <th>Status</th>
                                        <th>Actions</th>
                                    </tr>
                                </thead>
                                <tbody id="modal-table-body"></tbody>
                            </table>
                        </div>
                    </div>
                </div>
            </div>`;
        DOMElements.modalContainer.innerHTML = modalHTML;
        
        const modalTableBody = document.getElementById('modal-table-body');
        modalTableBody.innerHTML = '';
        if (issuesToShow.length === 0) {
            modalTableBody.innerHTML = '<tr><td colspan="7" style="text-align:center; color: var(--text-light);">No issues found</td></tr>';
            return;
        }
        
        issuesToShow.forEach(issue => {
            const priority = getPriorityLevel(issue.priorityScore);
            const severity = issue.severity || 1;
            const isCritical = issue.is_critical || severity === 3;
            
            const row = document.createElement('tr');
            if (isCritical) {
                row.style.backgroundColor = 'rgba(124, 58, 237, 0.1)';
            }
            
            row.innerHTML = `
                <td><span class="priority-dot ${priority.class}"></span>${priority.label} (${issue.priorityScore})</td>
                <td>${issue.title}</td>
                <td>${issue.category}</td>
                <td>
                    <div class="severity-indicator">
                        <span class="severity-pill severity-${severity}">${getSeverityLabel(severity)}</span>
                        ${isCritical ? '<i class="fas fa-exclamation-triangle" style="color: var(--danger-color);"></i>' : ''}
                    </div>
                </td>
                <td>${issue.upvotes}</td>
                <td>${issue.status}</td>
                <td><button class="btn view-details" data-issue-id="${issue._id || issue.id}">Details</button></td>`;
            modalTableBody.appendChild(row);
        });
    }

    async function loadIssues() {
        try {
            appState.issues = await API.authority.getIssues();
            return appState.issues;
        } catch (error) {
            console.error('Error loading issues:', error);
            return [];
        }
    }

    async function renderDashboard() {
        await loadIssues();
        
        // Calculate stats client-side from the full issue list
        const allIssues = appState.issues;
        const activeIssuesForStats = allIssues.filter(i => i.status !== 'Resolved');
        const resolvedIssuesCount = allIssues.length - activeIssuesForStats.length;
        
        const criticalCount = activeIssuesForStats.filter(i => i.severity === 3 || i.is_critical).length;
        const urgentCount = activeIssuesForStats.filter(i => i.severity === 2).length;
        const totalActiveCount = activeIssuesForStats.length;

        DOMElements.criticalCount.textContent = criticalCount;
        DOMElements.urgentCount.textContent = urgentCount;
        DOMElements.totalCount.textContent = totalActiveCount;
        DOMElements.resolvedCount.textContent = resolvedIssuesCount;

        const issuesWithPriority = allIssues.map(issue => ({ 
            ...issue, 
            priorityScore: calculatePriority(issue) 
        }));
        
        const activeIssues = issuesWithPriority.filter(i => i.status !== 'Resolved');
        const sortedActive = activeIssues.sort((a, b) => {
            const aIsCritical = a.is_critical || a.severity === 3;
            const bIsCritical = b.is_critical || b.severity === 3;

            if (aIsCritical && !bIsCritical) return -1;
            if (!aIsCritical && bIsCritical) return 1;

            // If both are critical or both not, sort by severity then priority
            if (a.severity !== b.severity) {
                return (b.severity || 1) - (a.severity || 1);
            }
            return b.priorityScore - a.priorityScore;
        });
        
        // Apply severity filter if set
        const severityFilter = DOMElements.severityFilter.value;
        let filteredIssues = sortedActive;
        if (severityFilter !== 'all') {
            const severityValue = parseInt(severityFilter);
            filteredIssues = sortedActive.filter(issue => 
                issue.severity === severityValue || (severityValue === 3 && issue.is_critical)
            );
        }
        
        populateTable(DOMElements.issuesTableBody, filteredIssues);
    }

    async function renderResolved() {
        const issues = await API.authority.getIssues('Resolved');
        const issuesWithPriority = issues.map(issue => ({ 
            ...issue, 
            priorityScore: calculatePriority(issue) 
        }));
        
        // Apply severity filter if set
        const severityFilter = DOMElements.severityFilter.value;
        let filteredIssues = issuesWithPriority;
        if (severityFilter !== 'all') {
            const severityValue = parseInt(severityFilter);
            filteredIssues = issuesWithPriority.filter(issue => 
                issue.severity === severityValue || (severityValue === 3 && issue.is_critical)
            );
        }
        
        populateTable(DOMElements.resolvedTableBody, filteredIssues);
    }

    function renderMap() {
        setTimeout(async () => {
            if (appState.map) appState.map.remove();
            appState.map = L.map('map-container').setView([12.8585, 80.1800], 14);
            L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '&copy; OpenStreetMap' }).addTo(appState.map);

            const activeIssues = appState.issues.filter(i => i.status !== 'Resolved');
            const heatPoints = activeIssues.map(issue => {
                const lat = issue.coordinates?.lat || issue.lat;
                const lng = issue.coordinates?.lng || issue.lng;
                const intensity = (issue.severity || 1) * 10; // Scale by severity
                return [lat, lng, intensity];
            });

            if (heatPoints.length > 0) {
                L.heatLayer(heatPoints, { radius: 25, blur: 15, maxZoom: 17, max: 50.0 }).addTo(appState.map);
            }

            activeIssues.forEach(issue => {
                const lat = issue.coordinates?.lat || issue.lat;
                const lng = issue.coordinates?.lng || issue.lng;
                if (lat && lng) {
                    const marker = L.marker([lat, lng]).addTo(appState.map);
                    marker.bindPopup(`<b>${issue.title}</b><br/>Severity: ${getSeverityLabel(issue.severity || 1)}<br/>Status: ${issue.status}<br/>Upvotes: ${issue.upvotes}`)
                        .on('click', () => openDetailsModal(issue._id || issue.id));
                    
                    // Color marker based on severity
                    const color = issue.severity === 3 ? '#7c3aed' : 
                                  issue.severity === 2 ? '#ef4444' : 
                                  issue.severity === 1 ? '#f59e0b' : '#10b981';
                    
                    marker.setIcon(L.divIcon({
                        className: 'custom-marker',
                        html: `<div style="background-color: ${color}; width: 20px; height: 20px; border-radius: 50%; border: 2px solid white;"></div>`,
                        iconSize: [20, 20]
                    }));
                }
            });
        }, 100);
    }

    async function renderAnalytics() {
        try {
            const analytics = await API.authority.getAnalytics();
            
            const categoryCtx = document.getElementById('categoryChart').getContext('2d');
            const resolutionCtx = document.getElementById('resolutionChart').getContext('2d');
            
            if (appState.charts.category) appState.charts.category.destroy();
            appState.charts.category = new Chart(categoryCtx, {
                type: 'pie',
                data: {
                    labels: analytics.byCategory.map(c => c._id),
                    datasets: [{
                        data: analytics.byCategory.map(c => c.count),
                        backgroundColor: ['#ef4444', '#f97316', '#f59e0b', '#3b82f6', '#10b981', '#8b5cf6']
                    }]
                }
            });

            if (appState.charts.resolution) appState.charts.resolution.destroy();
            appState.charts.resolution = new Chart(resolutionCtx, {
                type: 'bar',
                data: {
                    labels: Object.keys(analytics.avgResolutionTime),
                    datasets: [{
                        label: 'Avg. Days',
                        data: Object.values(analytics.avgResolutionTime),
                        backgroundColor: '#2563eb'
                    }]
                },
                options: { scales: { y: { beginAtZero: true } } }
            });
        } catch (error) {
            console.error('Error loading analytics:', error);
        }
    }

    function openDetailsModal(issueId) {
        const issue = appState.issues.find(i => (i._id || i.id) == issueId);
        if (!issue) return;

        const departments = [
            'Public Works Department',
            'Electricity Board',
            'Sanitation Department',
            'Water Board',
            'Traffic Police',
            'Forestry Department'
        ];
        const deptOptions = departments.map(d => `<option value="${d}">${d}</option>`).join('');
        
        const modalHTML = `
            <div class="modal-overlay">
                <div class="modal-content">
                    <div class="modal-header">
                        <h2>Issue Details (ID: ${issue._id || issue.id})</h2><button class="modal-close">&times;</button>
                    </div>
                    <div class="modal-body">
                        <div class="issue-details-grid">
                            <div>
                                <h3>${issue.title}</h3>
                                <p><strong>Location:</strong> ${issue.location}</p>
                                <p><strong>Category:</strong> ${issue.category}</p>
                                <p><strong>Description:</strong> ${issue.description}</p>
                                <p><strong>Severity:</strong> 
                                    <span class="severity-pill severity-${issue.severity || 1}">${getSeverityLabel(issue.severity || 1)}</span>
                                    ${issue.is_critical ? '<i class="fas fa-exclamation-triangle" style="color: var(--danger-color);"></i>' : ''}
                                </p>
                                <p><strong>Upvotes:</strong> ${issue.upvotes}</p>
                                <img src="${issue.image}" alt="${issue.title}">
                                <div id="detail-map-container"></div>
                            </div>
                            <div>
                                <h4>Case Management</h4>
                                <div class="form-group">
                                    <label>Assign To:</label>
                                    <select id="assign-department">${deptOptions}</select>
                                </div>
                                <div class="form-group">
                                    <label>Update Status:</label>
                                    <select id="update-status">
                                        <option value="Reported" ${issue.status === 'Reported' ? 'selected' : ''}>Reported</option>
                                        <option value="In Progress" ${issue.status === 'In Progress' ? 'selected' : ''}>In Progress</option>
                                        <option value="Resolved" ${issue.status === 'Resolved' ? 'selected' : ''}>Resolved</option>
                                    </select>
                                </div>
                                <div class="form-group">
                                    <label>Send Update to Citizen:</label>
                                    <textarea id="citizen-update" rows="4" placeholder="Enter message to citizen..."></textarea>
                                </div>
                                <button class="btn" id="save-changes" data-issue-id="${issue._id || issue.id}">Save & Notify Citizen</button>
                            </div>
                        </div>
                    </div>
                </div>
            </div>`;

        setTimeout(() => {
            const lat = issue.coordinates?.lat || issue.lat;
            const lng = issue.coordinates?.lng || issue.lng;
            if (lat && lng) {
                const detailMap = L.map('detail-map-container').setView([lat, lng], 16);
                L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(detailMap);
                L.marker([lat, lng]).addTo(detailMap);
            }
        }, 100);
    }

    async function handleLogin(e) {
        e.preventDefault();
        try {
            const { authority, token } = await API.authority.login({
                officialId: document.getElementById('officialId').value,
                password: document.getElementById('password').value
            });
            
            setAuthToken(token);
            appState.currentAuthority = authority;
            DOMElements.officialName.textContent = authority.name;
            DOMElements.loginContainer.style.display = 'none';
            DOMElements.appWrapper.style.display = 'block';
            await renderDashboard();
            showToast('Login successful!');
        } catch (error) {
            showToast(error.message, 'error');
        }
    }

    function handleLogout() {
        clearAuthToken();
        appState.currentAuthority = null;
        DOMElements.loginContainer.style.display = 'flex';
        DOMElements.appWrapper.style.display = 'none';
    }

    function switchView(viewId) {
        DOMElements.navLinks.forEach(l => l.classList.remove('active'));
        document.querySelector(`nav a[data-view="${viewId}"]`)?.classList.add('active');
        DOMElements.views.forEach(v => v.classList.remove('active'));
        document.getElementById(`${viewId}-view`)?.classList.add('active');

        if (viewId === 'map') renderMap();
        if (viewId === 'analytics') renderAnalytics();
        if (viewId === 'resolved') renderResolved();
    }

    // Event Listeners
    document.addEventListener('click', async (e) => {
        if (e.target.matches('.modal-close') || e.target.matches('.modal-overlay')) {
            DOMElements.modalContainer.innerHTML = '';
        }
        
        if (e.target.matches('.view-details')) {
            openDetailsModal(e.target.dataset.issueId);
        }
        
        if (e.target.id === 'save-changes') {
            const issueId = e.target.dataset.issueId;
            const status = document.getElementById('update-status').value;
            const message = document.getElementById('citizen-update').value;
            
            try {
                await API.authority.updateStatus(issueId, { status, message });
                showToast('Issue updated successfully!');
                DOMElements.modalContainer.innerHTML = '';
                await renderDashboard();
            } catch (error) {
                showToast(error.message, 'error');
            }
        }

        if (e.target.closest('.stat-card.clickable')) {
            const card = e.target.closest('.stat-card.clickable');
            if (card.dataset.view) {
                switchView(card.dataset.view);
            } else if (card.dataset.filter) {
                openFilteredIssueListModal(card.dataset.filter);
            }
        }
    });

    DOMElements.navLinks.forEach(link => {
        link.addEventListener('click', e => {
            e.preventDefault();
            switchView(link.dataset.view);
        });
    });

    DOMElements.darkModeToggle.addEventListener('click', () => {
        document.body.classList.toggle('dark-mode');
        const icon = DOMElements.darkModeToggle.querySelector('i');
        const text = DOMElements.darkModeToggle.querySelector('span');
        if (document.body.classList.contains('dark-mode')) {
            icon.className = 'fas fa-sun';
            text.textContent = 'Light Mode';
        } else {
            icon.className = 'fas fa-moon';
            text.textContent = 'Dark Mode';
        }
    });

    DOMElements.loginForm.addEventListener('submit', handleLogin);
    DOMElements.logoutBtn.addEventListener('click', handleLogout);

    // Severity filter event listener
    DOMElements.severityFilter.addEventListener('change', () => {
        renderDashboard();
    });

    // Auto-login if token exists
    const token = getAuthToken();
    if (token) {
        DOMElements.loginContainer.style.display = 'none';
        DOMElements.appWrapper.style.display = 'block';
        renderDashboard();
    }
})();
